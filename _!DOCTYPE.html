for i in range(0, 2 + 1, 1):
    lista[i] = int(input())
for i in range(0, 2 + 1, 1):
    risultato = 1
    for k in range(1, lista[i] + 1, 1):
        risultato = risultato * k
    fattoriali[i] = risultato
    print(fattoriali[i])
